 //Crear el algoritmo que multiplique dos numeros , sin usar el simbolo de multiplicar

function mult(a, b) {
    let resultado = 0;
    for (let index = 1; index <= b; index++) {
        resultado = resultado + a;
        resultado = 0
        resultado += a
    }
    return resultado
}
